    0.11\fraccorche{\$}{c}X_{22}\corche{c} +
    0.04\fraccorche{\$}{c}X_{23}\corche{c} -
    0.08\fraccorche{\$}{c}X_{24}\corche{c}

\begin{align*}
   X_{11}\corche{c} + X_{12}\corche{c} +    X_{13}\corche{c} + X_{14}\corche{c} \le 5000\corche{c} \\

\end{align*}
