\begin{align*}
   Min\ Z = \sum_{i=1}^{6}{X_i} 
\end{align*}

    1X_1+1X_2+0X_3+0X_4+1X_5+1X_6 \geq 1\\
    1X_1+0X_2+1X_3+1X_4+0X_5+1X_6 \geq 1\\
    0X_1+0X_2+1X_3+1X_4+1X_5+0X_6 \geq 1\\
